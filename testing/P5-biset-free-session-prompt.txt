## Ajouter le bi-set dans la séance libre (SessionBuilderModal)

### Contexte
Le composant `BiSetToggle` et la logique `toggleBiSet` / `isLinkedToNext` existent déjà
dans l'éditeur de programme coach (`src/components/coach/BiSetToggle.tsx`,
`src/lib/superset-utils.ts`). La table `session_exercises` possède déjà la colonne
`superset_group` (integer nullable). Il faut simplement les brancher sur la séance libre.

### Fichier 2 : `src/lib/superset-utils.ts`

Rends les deux fonctions génériques pour qu'elles acceptent aussi `LiveExercise[]` :

```typescript
type WithSupersetGroup = { supersetGroup?: number | null };

export function toggleBiSet<T extends WithSupersetGroup>(
  exercises: T[],
  index: number,
): T[] {
  const a = exercises[index];
  const b = exercises[index + 1];
  if (!a || !b) return exercises;

  const alreadyLinked =
    a.supersetGroup != null && a.supersetGroup === b.supersetGroup;

  if (alreadyLinked) {
    return exercises.map((e, i) =>
      i === index || i === index + 1 ? { ...e, supersetGroup: null } : e,
    );
  }

  const used = new Set<number>();
  for (const e of exercises) {
    if (typeof e.supersetGroup === "number") used.add(e.supersetGroup);
  }
  let g = 1;
  while (used.has(g)) g++;

  return exercises.map((e, i) =>
    i === index || i === index + 1 ? { ...e, supersetGroup: g } : e,
  );
}

export function isLinkedToNext<T extends WithSupersetGroup>(
  exercises: T[],
  index: number,
): boolean {
  const a = exercises[index];
  const b = exercises[index + 1];
  return !!(a && b && a.supersetGroup != null && a.supersetGroup === b.supersetGroup);
}
```
